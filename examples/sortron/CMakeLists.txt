add_custom_command(OUTPUT ${CMAKE_CURRENT_BINARY_DIR}/sortron.s PRE_LINK
  COMMAND ${CC65} -O -T -t none --cpu 65c02 -I ${CMAKE_CURRENT_SOURCE_DIR}/../../include ${CMAKE_CURRENT_SOURCE_DIR}/sortron.c -o ${CMAKE_CURRENT_BINARY_DIR}/sortron.s --codesize 100
  COMMENT "Compile sortron.c")
add_custom_command(OUTPUT ${CMAKE_CURRENT_BINARY_DIR}/sortron.o PRE_LINK
  COMMAND ${CA65} -t none --cpu 65c02 ${CMAKE_CURRENT_BINARY_DIR}/sortron.s -o ${CMAKE_CURRENT_BINARY_DIR}/sortron.o
  DEPENDS ${CMAKE_CURRENT_BINARY_DIR}/sortron.s
  COMMENT "Assemble sortron.o")
add_custom_command(OUTPUT ${CMAKE_CURRENT_BINARY_DIR}/sortron.img PRE_LINK
  COMMAND ${LD65} -m ${CMAKE_CURRENT_BINARY_DIR}/sortron.map -C ${CC65_DIR}/cfg/rpc8e.cfg  --obj ${CMAKE_CURRENT_BINARY_DIR}/sortron.o  --lib ${CMAKE_CURRENT_BINARY_DIR}/../../src/nrpck_display.lib  --lib ${CMAKE_CURRENT_BINARY_DIR}/../../src/nrpck_reboot.lib --lib ${CMAKE_CURRENT_BINARY_DIR}/../../src/nrpck_device.lib --lib ${CMAKE_CURRENT_BINARY_DIR}/../../src/nrpck.lib --lib ${CC65_DIR}/lib/sprintf.lib --lib ${CC65_DIR}/lib/math.lib --lib ${CC65_DIR}/lib/errno.lib --lib ${CMAKE_CURRENT_BINARY_DIR}/../../src/nrpck_drv_console.lib --lib ${CMAKE_CURRENT_BINARY_DIR}/../../src/nrpck_drv_ccmonitor.lib --lib ${CMAKE_CURRENT_BINARY_DIR}/../../src/nrpck_drv_matrix.lib --lib ${CMAKE_CURRENT_BINARY_DIR}/../../src/nrpck_drv_sortron.lib --lib ${CMAKE_CURRENT_BINARY_DIR}/../../src/nrpck_drv_ioexpander.lib --lib ${CMAKE_CURRENT_BINARY_DIR}/../../src/nrpck_drv_disk.lib --lib ${CC65_DIR}/lib/rpc8e.lib -o sortron.img
  COMMAND ${ALIGN} ${CMAKE_CURRENT_BINARY_DIR}/sortron.img ${CMAKE_CURRENT_BINARY_DIR}/sortron.map
  DEPENDS ${CMAKE_CURRENT_BINARY_DIR}/sortron.o ${CMAKE_CURRENT_BINARY_DIR}/../../src/nrpck_display.lib ${CMAKE_CURRENT_BINARY_DIR}/../../src/nrpck_device.lib ${CMAKE_CURRENT_BINARY_DIR}/../../src/nrpck_reboot.lib ${CMAKE_CURRENT_BINARY_DIR}/../../src/nrpck.lib ${CMAKE_CURRENT_BINARY_DIR}/../../src/nrpck_drv_console.lib ${CMAKE_CURRENT_BINARY_DIR}/../../src/nrpck_drv_matrix.lib ${CMAKE_CURRENT_BINARY_DIR}/../../src/nrpck_drv_sortron.lib ${CMAKE_CURRENT_BINARY_DIR}/../../src/nrpck_drv_ioexpander.lib ${CMAKE_CURRENT_BINARY_DIR}/../../src/nrpck_drv_disk.lib
  COMMENT "Link sortron.img"
  VERBATIM)
